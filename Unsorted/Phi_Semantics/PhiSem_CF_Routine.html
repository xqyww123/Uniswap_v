Theory PhiSem_CF_Routine

theory PhiSem_CF_Routine
  imports PhiSem_CF_Break PhiSem_CF_Routine_Basic
begin

definition op_routine
        :: ‹TY list ⇒ TY list ⇒
            (RES.brk_label ⇒ ('a::FIX_ARITY_VALs, 'b::FIX_ARITY_VALs) proc') ⇒
            ('a,'b) proc'›
  where ‹op_routine argtys rettys F =
    op_routine_basic argtys rettys (λarg. op_brk_scope (λbrk. F brk arg))›

abbreviation
  ‹op_rec_routine argtys rettys F ≡ op_fix_point (λ𝖿. op_routine argtys rettys (F 𝖿))›

lemma "__routine__":
  ‹ φ_Have_Types X TY_ARGs
⟹ φ_Have_Types Y TY_RETs
⟹ 𝗋Success
⟹ (⋀(vs:: 'a::FIX_ARITY_VALs φarg <named> 'names) label_ret.
      return_φapp❙: TECHNICAL(
          ∀ret :: 'b :: FIX_ARITY_VALs φarg.
            𝗉𝗋𝗈𝖼 (op_break label_ret ret :: 'b proc) ⦃ Y ret❟ TECHNICAL Brk_Frame label_ret ⟼ 0 ⦄
            𝗍𝗁𝗋𝗈𝗐𝗌 (λ_. Brking_Frame label_ret Y))
      ⟹ 𝗉𝗋𝗈𝖼 F label_ret (case_named (λx. x) vs) ⦃ X (case_named id vs)❟ TECHNICAL Brk_Frame label_ret
                                          ⟼ λret. Y ret❟ TECHNICAL Brk_Frame label_ret
                                            ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 (λe. ❙b❙r❙e❙a❙k label_ret ❙w❙i❙t❙h Y ❙o❙r E e))
⟹ 𝗉𝗋𝗈𝖼 op_routine TY_ARGs TY_RETs F vs ⦃ X vs ⟼ Y ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E›
  unfolding op_routine_def
  ❴ premises [φreason 5000] and [φreason 5000] and _ and F
    "__routine_basic__"[where TY_ARGs=TY_ARGs and TY_RETs=TY_RETs and X=X and Y=Y and vs=vs and 'names='names, simplified]
    ❴ for vs
      brk_scope ❴ for label_ret
        F[where vs=‹tag vs›, unfolded Technical_def[where 'a=‹bool›], simplified]
        "_op_break_rule_"
      ❵. ;;
    ❵. ;;
  ❵. .

attribute_setup routine =
  ‹Scan.succeed (Phi_Modifier.wrap_to_attribute
      (PhiSem_Control_Flow.routine_mod
          (fn T => \<^typ>‹RES.brk_label› --> T)
          (fn N => fn wrap => fn Tm => Abs("label_ret", \<^typ>‹RES.brk_label›, wrap (Tm $ Bound N)))
          @{thm "__routine__"}))›


end
