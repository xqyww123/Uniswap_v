definition φM_get_var :: "varname ⇒ TY ⇒ (VAL ⇒ 'ret proc) ⇒ 'ret::VALs proc"
  where "φM_get_var vname TY F = RES.Var.φR_get_res_entry vname ((λval.
            φM_assert (val ∈ Some ` Well_Type TY) ≫ F (the val)) o nosep.dest)"

definition op_get_var :: "varname ⇒ TY ⇒ VAL proc"
  where "op_get_var vname TY = φM_get_var vname TY (λx. Return (φarg x))"

definition op_set_var :: "varname ⇒ TY ⇒ (VAL,unit) proc'"
  where "op_set_var vname TY v =
          φM_assert (pred_option (λTY'. TY = TY') (varname.type vname)) ≫
          φM_getV TY id v (λv.
          RES.Var.φR_set_res (λf. f(vname := Some (nosep (Some v)))))"

definition op_free_var :: "varname ⇒ unit proc"
  where "op_free_var vname = RES.Var.φR_set_res (λf. f(vname := None))"

definition op_var_scope' :: "TY option ⇒ (varname ⇒ 'ret proc) ⇒ 'ret::VALs proc"
  where "op_var_scope' TY F =
    RES.Var.φR_allocate_res_entry (λv. varname.type v = TY) (Some (nosep None)) F"

lemma φM_get_var[intro!]:
  ‹ v ∈ Well_Type TY
⟹ 𝗉𝗋𝗈𝖼 F v ⦃ v ⦂ 𝗏𝖺𝗋[var] Identity ⟼ Y ⦄
⟹ 𝗉𝗋𝗈𝖼 φM_get_var var TY F ⦃ v ⦂ 𝗏𝖺𝗋[var] Identity ⟼ Y ⦄›
  unfolding φM_get_var_def Inited_Var_identity_eq
  by (rule FIC.Var.φR_get_res_entry; simp)

lemma φM_set_var[intro!]:
  ‹𝗉𝗋𝗈𝖼 RES.Var.φR_set_res (λf. f(vname ↦ nosep (Some u)))
      ⦃ v ⦂ Var vname Identity ⟼ λ_. u ⦂ 𝗏𝖺𝗋[vname] Identity ⦄›
  unfolding Inited_Var_identity_eq
  unfolding Raw_Var_identity_eq
  thm FIC.Var.φR_set_res
  by (rule FIC.Var.φR_set_res[where P=‹λ_. True›]; simp)

   ‹𝗉𝗋𝖾𝗆𝗂𝗌𝖾 v ∈ Well_Type TY
