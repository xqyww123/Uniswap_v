Theory PhiSem_CF_Solidity

theory PhiSem_CF_Solidity
  imports PhiSem_CF_Basic
begin

(*for now we use a quick-and-dirty formalization of the semantics that doesn't use exception mechanism*)

definition op_require :: ‹(VAL, unit) proc'›
  where ‹op_require rP =
    φM_getV bool V_bool.dest rP φM_assert›
 
lemma require_φapp:
  ‹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 P
⟹ 𝗉𝗋𝗈𝖼 op_require rP ⦃ P ⦂ 𝗏𝖺𝗅[rP] 𝔹 ⟼ λ_. Void ⦄›
  unfolding op_require_def
  by (cases rP; simp, rule, simp add: φexpns WT_bool, blast, rule, simp add: φexpns WT_bool)


definition op_require' :: ‹(VAL, unit) proc'›
  where ‹op_require' rP =
    φM_getV bool V_bool.dest rP φM_assume›

lemma op_require'_φapp:
  ‹𝗉𝗋𝗈𝖼 op_require' rP ⦃ P ⦂ 𝗏𝖺𝗅[rP] 𝔹 ⟼ λ_. Void 𝗌𝗎𝖻𝗃 P ⦄›
  unfolding op_require'_def
  by (cases rP; simp, rule, simp add: φexpns WT_bool, simp add: φexpns WT_bool, rule)

end
