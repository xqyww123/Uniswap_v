Theory PhiSem_Generic_Boolean

chapter ‹Generic Boolean›

theory PhiSem_Generic_Boolean
  imports Phi_System.PhiSem_Formalization_Tools
begin

section ‹Semantic›

consts bool :: TY

debt_axiomatization V_bool :: ‹bool value_entry›
  where V_bool_ax: ‹VDT_field V_bool›

interpretation V_bool: VDT_field V_bool using V_bool_ax .

debt_axiomatization
      can_eq_bool: ‹Can_EqCompare res (V_bool.mk x1) (V_bool.mk x2)›
  and eq_bool:     ‹EqCompare (V_bool.mk x1) (V_bool.mk x2) = (x1 = x2)›
  and zero_bool:   ‹Zero bool = Some (V_bool.mk False)›
  and WT_bool:     ‹Well_Type bool = { V_bool.mk x |x. True }›

section ‹Instructions›

definition op_const_bool :: "bool ⇒ VAL proc"
  where "op_const_bool b = Return (φarg (V_bool.mk b))"

definition op_not :: "(VAL, VAL) proc'"
  where "op_not v =
    φM_getV bool V_bool.dest v (λv.
    Return (φarg (V_bool.mk (¬ v)))
  )"

definition op_and :: "(VAL × VAL, VAL) proc'"
  where "op_and =
    φM_caseV (λva vb.
    φM_getV bool V_bool.dest va (λv.
    φM_getV bool V_bool.dest vb (λu.
    Return (φarg (V_bool.mk (v ∧ u)))
  )))"

definition op_or :: "(VAL × VAL, VAL) proc'"
  where "op_or =
    φM_caseV (λva vb.
    φM_getV bool V_bool.dest va (λv.
    φM_getV bool V_bool.dest vb (λu.
    Return (φarg (V_bool.mk (v ∨ u)))
  )))"

definition op_equal :: "TY ⇒ (VAL × VAL, VAL) proc'"
  where "op_equal TY =
    φM_caseV (λva vb.
    φM_getV TY id va (λv.
    φM_getV TY id vb (λu.
    (λres. φM_assert (Can_EqCompare res u v) res) ≫
    Return (φarg (V_bool.mk (EqCompare u v)))
)))"


section ‹φ-Type›

definition φBool :: "(VAL, bool) φ" ("𝔹")
  where "𝔹 x = { V_bool.mk x }"

lemma φBool_expn[φexpns]:
  " p ∈ (x ⦂ 𝔹) ⟷ p = V_bool.mk x"
  unfolding φType_def φBool_def by simp

lemma φBool_inhabited[φinhabitance_rule, elim!]:
  ‹Inhabited (x ⦂ 𝔹) ⟹ C ⟹ C› .

lemma φBool_eqcmp[φreason 2000]:
  "φEqual 𝔹 (λx y. True) (=)"
  unfolding φEqual_def by (simp add: φexpns can_eq_bool eq_bool)

lemma φBool_zero[φreason 2000]:
  "𝗉𝗋𝖾𝗆𝗂𝗌𝖾 TY = bool ⟹ φZero TY 𝔹 False"
  unfolding φZero_def by (simp add: φexpns zero_bool)

lemma φBool_semty[φreason 2000]:
  ‹φSemType (x ⦂ 𝔹) bool›
  unfolding φSemType_def subset_iff
  by (simp add: φexpns WT_bool)

lemma [φreason 2000]:
  ‹is_singleton (x ⦂ 𝔹)›
  by (rule is_singletonI''; simp add: φexpns)

abbreviation ‹Predicate_About x ≡ (𝔹 <func-over> x)›


section ‹Abstractions of Boolean Arithmetic›

φoverloads "=" and "¬" and "∧" and "∨"

subsection ‹Constant›

lemma op_const_bool_φapp[φsynthesis for ‹λv. True ⦂ ?T v› (1200) and ‹λv. False ⦂ ?T v› (1200)]:
  ‹ Is_Literal b
⟹ 𝗉𝗋𝗈𝖼 op_const_bool b ⦃ Void ⟼ 𝗏𝖺𝗅 b ⦂ 𝔹 ⦄›
  unfolding op_const_bool_def
  by (rule, simp add: φBool_expn)

lemma True_φapp:
  ‹𝗉𝗋𝗈𝖼 op_const_bool True ⦃ Void ⟼ 𝗏𝖺𝗅 True ⦂ 𝔹 ⦄›
  ❴ ‹True› ❵.

lemma False_φapp:
  ‹𝗉𝗋𝗈𝖼 op_const_bool False ⦃ Void ⟼ 𝗏𝖺𝗅 False ⦂ 𝔹 ⦄›
  ❴ ‹False› ❵.

subsection ‹Not›

lemma op_not[φoverload ¬, φsynthesis 100]:
  ‹𝗉𝗋𝗈𝖼 op_not raw ⦃ x ⦂ 𝗏𝖺𝗅[raw] 𝔹 ⟼ 𝗏𝖺𝗅 ¬ x ⦂ 𝔹 ⦄›
  unfolding op_not_def
  by (cases raw, simp, rule, simp add: φexpns WT_bool, rule, simp add: φexpns)


subsection ‹And›

lemma op_and[φoverload ∧, φsynthesis add]:
  ‹𝗉𝗋𝗈𝖼 op_and (φV_pair vb va) ⦃ a ⦂ 𝗏𝖺𝗅[va] 𝔹❟ b ⦂ 𝗏𝖺𝗅[vb] 𝔹 ⟼ 𝗏𝖺𝗅 (a ∧ b) ⦂ 𝔹 ⦄›
  unfolding op_and_def
  by (cases va; cases vb; simp, rule, rule, simp add: φexpns WT_bool, rule,
      simp add: φexpns WT_bool, rule, simp add: φexpns, blast)

subsection ‹Or›

lemma op_or[φoverload ∨, φsynthesis 100]:
  ‹𝗉𝗋𝗈𝖼 op_or (φV_pair vb va) ⦃ a ⦂ 𝗏𝖺𝗅[va] 𝔹❟ b ⦂ 𝗏𝖺𝗅[vb] 𝔹 ⟼ 𝗏𝖺𝗅 (a ∨ b) ⦂ 𝔹 ⦄›
  unfolding op_or_def
  by(cases va; cases vb, simp, rule, rule, simp add: φexpns WT_bool, rule,
      simp add: φexpns WT_bool, rule, simp add: φexpns, blast)


subsection ‹Equal›

declare [[
    overloaded_operator_in_synthesis ‹λv. x ⦂ T v› ‹λv. y ⦂ U v› ⇒ ‹λv. x = y ⦂ 𝗏𝖺𝗅[v] 𝔹›,
    overloaded_operator_in_synthesis
        ‹λv. x mod N ⦂ T v› ‹λv. y mod N ⦂ U v› ⇒ ‹λv. x mod N = y mod N ⦂ 𝗏𝖺𝗅[v] 𝔹›
]]

lemma op_equal_φapp[φoverload =]:
  ‹ φEqual T can_eq eq
⟹ φSemType (a ⦂ T) TY
⟹ φSemType (b ⦂ T) TY
⟹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 can_eq a b
⟹ 𝗉𝗋𝗈𝖼 op_equal TY (φV_pair rawb rawa) ⦃ a ⦂ 𝗏𝖺𝗅[rawa] T❟ b ⦂ 𝗏𝖺𝗅[rawb] T ⟼ eq a b ⦂ 𝗏𝖺𝗅 𝔹 ⦄›
  unfolding op_equal_def
  apply (cases rawa; cases rawb; simp, rule, rule)
    apply (simp add: φSemType_def subset_iff Premise_def, rule)
    apply (simp add: φSemType_def subset_iff Premise_def, rule)
   apply (unfold φEqual_def Premise_def, simp)
  by (rule φM_Success', rule, simp add: φexpns)


declare op_equal_φapp[where eq=‹(=)›, φsynthesis 100]
declare op_equal_φapp[where eq=‹(λx y. x mod N = y mod N)› for N, φsynthesis 100]


end
