Theory PhiSem_Aggregate_Array

theory PhiSem_Aggregate_Array
  imports PhiSem_Aggregate_Base
begin

section ‹Semantics›

subsection ‹Models›

subsubsection ‹Type›

virtual_datatype array_ty =
  array   :: ‹'self × nat›

debt_axiomatization T_array :: ‹(TY × nat) type_entry›
  where array_ty_ax: ‹array_ty TY_CONS_OF T_array›

interpretation array_ty TY_CONS_OF ‹TYPE(TY_N)› ‹TYPE(TY)› T_array using array_ty_ax .

hide_fact array_ty_ax

abbreviation ‹array N T ≡ array.mk (T,N)›


subsubsection ‹Value›

virtual_datatype array_val =
  V_array   :: ‹'self list›

debt_axiomatization V_array :: ‹(VAL list) value_entry›
  where array_val_ax: ‹array_val VAL_CONS_OF V_array›

interpretation array_val VAL_CONS_OF ‹TYPE(VAL_N)› ‹TYPE(VAL)› V_array using array_val_ax .

hide_fact array_val_ax


subsection ‹Semantics›

debt_axiomatization
        WT_arr[simp]: ‹Well_Type (array n t) = { V_array.mk vs |vs. length vs = n ∧ list_all (λv. v ∈ Well_Type t) vs }›
  and   zero_arr[simp]: ‹Zero (array N T)  = map_option (λz. V_array.mk (replicate N z)) (Zero T)›
  and   idx_step_type_arr [eval_semantic_index] : ‹i ≤ N ⟹ idx_step_type i (array N T) = T›
  and   valid_idx_step_arr[eval_semantic_index] : ‹valid_idx_step (array N T) i ⟷ i < N›
  and   idx_step_value_arr[eval_semantic_index] : ‹idx_step_value i (V_array.mk vs) = vs!i›
  and   idx_step_mod_value_arr : ‹idx_step_mod_value i f (V_array.mk vs) = V_array.mk (vs[i := f (vs!i)])›

lemma list_all_replicate:
  ‹list_all P (replicate n x) ⟷ n = 0 ∨ P x›
  by (induct n; simp; blast)

(*lemma Valid_Type_τArray[simp]:
  ‹Valid_Type (array n T) ⟷ Valid_Type T›
  by (simp add: Inhabited_def;

section ‹φType›

definition Array :: "nat ⇒ (VAL, 'a) φ ⇒ (VAL, 'a list) φ"
  where ‹Array N T = (λl. { V_array.mk vs |vs. length l = N ∧ list_all2 (λv x. v ∈ (x ⦂ T)) vs l  })›

lemma Array_expns[φexpns]:
  ‹v ∈ (l ⦂ Array N T) ⟷
    (∃vs. length l = N ∧ v = V_array.mk vs ∧ list_all2 (λv x. v ∈ (x ⦂ T)) vs l )›
  unfolding Array_def φType_def by simp blast

lemma Array_inhabited[φinhabitance_rule, elim!]:
  ‹ Inhabited (l ⦂ Array N T)
⟹ (length l = N ⟹ (⋀i. i < length l ⟹ Inhabited (l!i ⦂ T)) ⟹ C)
⟹ C›
  unfolding Inhabited_def by (clarsimp simp add: φexpns list_all2_conv_all_nth) blast

lemma Array_semty[φreason 1000]:
  ‹(⋀x. φSemType (x ⦂ T) TY) ⟹ φSemType (x ⦂ Array N T) (array N TY)›
  by (clarsimp simp add: φexpns list_all_length list_all2_conv_all_nth φSemType_def subset_iff
          Inhabited_def, blast)

lemma Array_zero[φreason 1000]:
  ‹φZero TY T zero ⟹ φφSemType T TY ⟹ φZero (array N TY) (Array N T) (replicate N zero)›
  unfolding φZero_def
  by (clarsimp simp add: φexpns list_all2_conv_all_nth Inhabited_def image_iff; blast)


section ‹Reasoning›

subsection ‹Index to Fields of Structures›


lemma [φreason 1200]:
  ‹ φIndex_getter idx X Y f
⟹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 i < N
⟹ φIndex_getter (i # idx) (Array N X) Y (λl. f (l!i))›
  unfolding φIndex_getter_def Premise_def
  by (clarsimp simp add: φexpns idx_step_value_arr list_all2_conv_all_nth)

lemma [φreason 1200]:
  ‹ φIndex_mapper idx X X Y Y' f
⟹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 i < N
⟹ φIndex_mapper (i # idx) (Array N X) (Array N X) Y Y' (λg l. l[i := f g (l!i)])›
  unfolding φIndex_mapper_def Premise_def
  by (clarsimp simp add: φexpns idx_step_mod_value_arr list_all2_conv_all_nth nth_list_update)
