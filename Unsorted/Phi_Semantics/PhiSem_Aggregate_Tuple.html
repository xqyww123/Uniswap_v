section ‹Semantics›

subsection ‹Models›

subsubsection ‹Type›

virtual_datatype tuple_ty =
  tup     :: ‹'self list›

debt_axiomatization T_tup :: ‹TY list type_entry›
  where tuple_ty_ax: ‹tuple_ty TY_CONS_OF T_tup›

interpretation tuple_ty TY_CONS_OF ‹TYPE(TY_N)› ‹TYPE(TY)› T_tup using tuple_ty_ax .

(*TODO: intergrate the hidding into the automation command*)
hide_fact tuple_ty_ax

subsubsection ‹Value›

virtual_datatype tuple_val =
  V_tup     :: ‹'self list›

debt_axiomatization V_tup :: ‹VAL list value_entry›
  where tuple_val_ax: ‹tuple_val VAL_CONS_OF V_tup›

interpretation tuple_val VAL_CONS_OF ‹TYPE(VAL_N)› ‹TYPE(VAL)› V_tup
  using tuple_val_ax .

hide_fact tuple_val_ax

abbreviation ‹tup ≡ tup.mk›

subsection ‹Semantics›

debt_axiomatization
        WT_tup[simp]: ‹Well_Type (tup ts)  = { V_tup.mk vs       |vs. list_all2 (λ t v. v ∈ Well_Type t) ts vs }›
  and   zero_tup[simp]: ‹Zero (tup Ts)     = map_option V_tup.mk (those (map Zero Ts))›
  and   V_tup_sep_disj_R[simp]: ‹V_tup.mk l1 ## vl2 ⟷ (∃l2. vl2 = V_tup.mk l2)›
  and   V_tup_sep_disj_L[simp]: ‹vl1 ## V_tup.mk l2 ⟷ (∃l1. vl1 = V_tup.mk l1)›
  and   V_tup_mult    : ‹V_tup.mk l1 * V_tup.mk l2 = V_tup.mk (l2 @ l1)›
  and   idx_step_type_tup [eval_semantic_index] : ‹i < length tys ⟹ idx_step_type i (tup tys) = tys!i ›
  and   valid_idx_step_tup[eval_semantic_index] : ‹valid_idx_step (tup tys) i ⟷ i < length tys›
  and   idx_step_value_tup[eval_semantic_index] : ‹idx_step_value i (V_tup.mk vs) = vs!i›
  and   idx_step_mod_value_tup : ‹idx_step_mod_value i f (V_tup.mk vs) = V_tup.mk (vs[i := f (vs!i)])›

lemma V_tup_mult_cons:
  ‹NO_MATCH vs ([]::VAL list) ⟹ V_tup.mk (v#vs) = V_tup.mk vs * V_tup.mk [v]›
  using V_tup_mult by simp

lemma list_all_replicate:
  ‹list_all P (replicate n x) ⟷ n = 0 ∨ P x›
  by (induct n; simp; blast)

section ‹φType›

subsection ‹Empty Tuple›

definition EmptyTuple :: ‹(VAL, unit) φ›
  where ‹EmptyTuple x = { V_tup.mk [] }›

lemma EmptyTuple_expn[φexpns]:
  ‹p ∈ (x ⦂ EmptyTuple) ⟷ p = V_tup.mk []›
  unfolding EmptyTuple_def φType_def by simp

subsection ‹Field›

definition Tuple_Field :: "(VAL, 'a) φ ⇒ (VAL, 'a) φ"
  where "Tuple_Field T x = { V_tup.mk [v] |v. v ∈ T x }"

syntax "_φTuple" :: ‹tuple_args ⇒ logic› ("⦃ _ ⦄")

translations
  "_φTuple (_tuple_arg X)" ⇌ "CONST Tuple_Field X"
  "_φTuple (_tuple_args y z)" ⇌ "CONST φProd (_φTuple (_tuple_arg y)) (_φTuple z)"

lemma Tuple_Field_expn[φexpns]:
  ‹p ∈ (x ⦂ ⦃ T ⦄) ⟷ (∃v. p = V_tup.mk [v] ∧ v ∈ (x ⦂ T))›
  unfolding Tuple_Field_def φType_def by simp

lemma Tuple_Field_inhabited[elim!,φinhabitance_rule]:
  ‹Inhabited (x ⦂ ⦃ T ⦄) ⟹ (Inhabited (x ⦂ T) ⟹ C) ⟹ C›
  unfolding Inhabited_def by (simp add: φexpns)

lemma EmptyTuple_reduce[simp]:
  ‹(((),a) ⦂ EmptyTuple ∗ ⦃ T ⦄) = (a ⦂ ⦃ T ⦄)›
  ‹((a,()) ⦂ ⦃ T ⦄ ∗ EmptyTuple) = (a ⦂ ⦃ T ⦄)›
  unfolding set_eq_iff
  apply (clarsimp; rule; clarsimp simp add: φexpns V_tup_mult)
  apply (metis V_tup_mult append_Nil)
  apply (clarsimp; rule; clarsimp simp add: φexpns V_tup_mult)
  by (metis V_tup_mult append.right_neutral)

lemma Tuple_Field_zero  [φreason 1000]:
  ‹φZero ty T x ⟹ φZero (tup [ty]) ⦃ T ⦄ x ›
  unfolding φZero_def by (clarsimp simp add: φexpns)

lemma Tuple_Field_zeros [φreason 1000]:
  ‹φZero ty T x
    ⟹ φZero (tup tys) Ts xs
    ⟹ φZero (tup (ty#tys)) (⦃ T ⦄ ∗ Ts) (x,xs) ›
  unfolding φZero_def
  apply (clarsimp simp add: φexpns V_tup_mult_cons image_iff)
  using V_tup_sep_disj_L by blast

lemma Tuple_Field_semty[φreason 1000]:
  ‹ φSemType (x ⦂ T) TY
⟹ φSemType (x ⦂ ⦃ T ⦄) (tup [TY])›
  unfolding φSemType_def subset_iff
  by (clarsimp simp add: φexpns)

lemma Tuple_Field_semtys[φreason 1000]:
  ‹ φSemType (x ⦂ T) TY
⟹ φSemType (xs ⦂ Ts) (tup TYs)
⟹ φSemType ((x,xs) ⦂ (⦃ T ⦄ ∗ Ts)) (tup (TY#TYs))›
  unfolding φSemType_def subset_iff
  apply (clarsimp simp add: φexpns)
  by (metis V_tup_mult append.left_neutral append_Cons list.rel_inject(2))

  by (metis NO_MATCH_I V_tup_mult_cons idx_step_mod_value_tup list_update_code(3) nth_Cons_Suc)
