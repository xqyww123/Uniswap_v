definition op_const_areal :: "real ⇒ VAL proc"
  where "op_const_areal const = Return (φarg (V_areal.mk const))"

definition op_add_ar :: "(VAL × VAL, VAL) proc'"
  where "op_add_ar =
      φM_caseV (λvb va.
      φM_getV areal V_areal.dest vb (λval_b.
      φM_getV areal V_areal.dest va (λval_a.
      Return (φarg (V_areal.mk (val_a + val_b)))
  )))"

definition op_sub_ar :: "(VAL × VAL, VAL) proc'"
  where "op_sub_ar =
      φM_caseV (λva vb.
      φM_getV areal V_areal.dest va (λval_a.
      φM_getV areal V_areal.dest vb (λval_b.
      Return (φarg (V_areal.mk (val_b - val_a)))
  )))"

definition op_neg_ar :: "(VAL, VAL) proc'"
  where "op_neg_ar rv =
      φM_getV areal V_areal.dest rv (λv.
      Return (φarg (V_areal.mk (-v))))"

definition op_mul_ar :: "(VAL × VAL, VAL) proc'"
