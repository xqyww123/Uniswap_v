lemma overloaded_synthesis_const:
  ‹OPTIMAL_SYNTHESIS
   (𝗉𝗋𝗈𝖼 H ⦃ R ⟼ λret. R❟ ❬ MK_CONST const ⦂ T ret ❭ ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E @action overloaded_synthesis)
⟹ 𝗉𝗋𝗈𝖼 H ⦃ R ⟼ λret. R❟ ❬ const ⦂ T ret ❭ ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E @action synthesis›
  unfolding Optimal_Synthesis_def Action_Tag_def MK_CONST_def .

lemma make_overloaded_synthesis_rule_for_const:
  ‹ Simplify (assertion_simps ABNORMAL) E' (λe. R❟ E e)
⟹ PROP Gen_Synthesis_Rule
          (Trueprop (∀vs::unit φarg. 𝗉𝗋𝗈𝖼 F vs ⦃ X vs ⟼ λret. const ⦂ T ret ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E))
          Ant
          (X' 𝗂𝗆𝗉𝗅𝗂𝖾𝗌 X φV_none 𝗋𝖾𝗆𝖺𝗂𝗇𝗌 R 𝖺𝗇𝖽 Any1
       ⟹ PROP Ant
       ⟹ 𝗉𝗋𝗈𝖼 F φV_none ⦃ X' ⟼ λret. R❟ ❬ MK_CONST const ⦂ T ret ❭ ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E'
           @action overloaded_synthesis)›
  unfolding MK_CONST_def split_paired_All_φarg_unit
  using make_overloaded_synthesis_rule[where 'a = ‹unit φarg› and X' = ‹λ_. X'›,
      unfolded split_paired_All_φarg_unit split_paired_all_φarg_unit] .
