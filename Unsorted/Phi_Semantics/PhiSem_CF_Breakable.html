Theory PhiSem_CF_Breakable

theory PhiSem_CF_Breakable
  imports PhiSem_CF_Break PhiSem_CF_Basic
begin


text ‹Since we have ▩‹break› and ▩‹continue› now, the termination condition of a loop is not
  necessarily the negative of the loop guard. Therefore here we need 3 assertions, invariance,
  guard, and termination condition.›

proc while:
  requires ‹𝗉𝖺𝗋𝖺𝗆 (X x 𝗌𝗎𝖻𝗃 x. Inv: invariant x ∧ Guard: cond x ∧ End: termination x)›
  and S: ‹U 𝗂𝗆𝗉𝗅𝗂𝖾𝗌 ((X x 𝗋𝖾𝗆𝖺𝗂𝗇𝗌 R) 𝗌𝗎𝖻𝗃 x. invariant x) @action ToSA›
  and ‹𝗉𝗋𝖾𝗆𝗂𝗌𝖾 (∀x. invariant x ∧ ¬ cond x ⟶ termination x)›
  and C: "⋀x. 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 invariant x ⟹
                  𝗉𝗋𝗈𝖼 Cond ⦃ R❟ X x ⟼ R❟ X x'❟ 𝗏𝖺𝗅 cond x' ⦂ 𝔹 𝗌𝗎𝖻𝗃 x'. invariant x' ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E1"
  and B: "⋀x lb lc. 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 invariant x ⟹
                    𝗉𝗋𝖾𝗆𝗂𝗌𝖾 cond x ⟹
                    break_φapp❙: TECHNICAL
                        𝗉𝗋𝗈𝖼 (op_break lb φV_none :: unit proc)
                           ⦃ (R❟ X x'❟ TECHNICAL Brk_Frame lc 𝗌𝗎𝖻𝗃 x'. invariant x' ∧ termination x')❟ TECHNICAL Brk_Frame lb ⟼ 0 ⦄
                        𝗍𝗁𝗋𝗈𝗐𝗌 (λ_. Brking_Frame lb (λ_::unit φarg. R❟ X x'❟ TECHNICAL Brk_Frame lc 𝗌𝗎𝖻𝗃 x'. invariant x' ∧ termination x')) ⟹
                    continue_φapp❙: TECHNICAL
                        𝗉𝗋𝗈𝖼 (op_break lc φV_none :: unit proc)
                           ⦃ (R❟ X x'❟ TECHNICAL Brk_Frame lb 𝗌𝗎𝖻𝗃 x'. invariant x')❟ TECHNICAL Brk_Frame lc ⟼ 0 ⦄
                        𝗍𝗁𝗋𝗈𝗐𝗌 (λ_. Brking_Frame lc (λ_::unit φarg. R❟ X x'❟ TECHNICAL Brk_Frame lb 𝗌𝗎𝖻𝗃 x'. invariant x')) ⟹
                    𝗉𝗋𝗈𝖼 Body lb lc
                        ⦃ R❟ X x ❟ TECHNICAL Brk_Frame lc❟ TECHNICAL Brk_Frame lb
                      ⟼ R❟ X x'❟ TECHNICAL Brk_Frame lc❟ TECHNICAL Brk_Frame lb 𝗌𝗎𝖻𝗃 x'. invariant x'
                         ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 (λe. ❙b❙r❙e❙a❙k lb ❙w❙i❙t❙h (λ_::unit φarg. R❟ X x'❟ TECHNICAL Brk_Frame lc 𝗌𝗎𝖻𝗃 x'. invariant x' ∧ termination x')
                                    ❙o❙r ❙b❙r❙e❙a❙k lc ❙w❙i❙t❙h (λ_::unit φarg. R❟ X x' 𝗌𝗎𝖻𝗃 x'. invariant x')
                                    ❙o❙r E2 e)"
  input ‹U›
  output ‹R❟ X x 𝗌𝗎𝖻𝗃 x. invariant x ∧ termination x›
  throws ‹E1 + E2›
  ❴ S
    brk_scope ❴ for lb
      PhiSem_CF_Basic.while ‹TECHNICAL Brk_Frame lb❟ R❟ X x 𝗌𝗎𝖻𝗃 x. Inv: invariant x ∧ Guard: cond x›
      ❴ C ❵
      ❴ brk_scope ❴ for lc
          B[where lb1=lb]
          "_op_break_rule_"[THEN Technical_I, THEN Labelled_I]
          "_op_break_rule_"[THEN Technical_I, THEN Labelled_I]
        ❵ for ‹(R❟ X x'❟ TECHNICAL Brk_Frame lb 𝗌𝗎𝖻𝗃 x'. invariant x')❟ TECHNICAL Brk_Frame lc› ;;
      ❵ ;;
    ❵ for ‹(R❟ X x' 𝗌𝗎𝖻𝗃 x'. invariant x' ∧ termination x')❟ TECHNICAL Brk_Frame lb› ;;
  ❵.

hide_const (open) PhiSem_CF_Basic.while

end
