theory PhiSem_Real_Abst_Int
  imports PhiSem_Real_Abst PhiSem_Int_ArbiPrec
begin

section ‹Conversion between Abstract Real and Arbitrary-precision Integers›

definition op_Ra_floor_Za :: ‹(VAL, VAL) proc'›
  where ‹op_Ra_floor_Za rv =
    φM_getV areal V_areal.dest rv (λv.
    Return (φarg (V_aint.mk ⌊v⌋)))›

definition op_Ra_ceiling_Za :: ‹(VAL, VAL) proc'›
  where ‹op_Ra_ceiling_Za rv =
    φM_getV areal V_areal.dest rv (λv.
    Return (φarg (V_aint.mk ⌈v⌉)))›

lemma op_Ra_floor_Za_φapp[φoverload floor, φsynthesis 100]:
  ‹𝗉𝗋𝗈𝖼 op_Ra_floor_Za raw ⦃ x ⦂ 𝗏𝖺𝗅[raw] ℝ ⟼ ⌊x⌋ ⦂ 𝗏𝖺𝗅 ℤ ⦄›
  unfolding op_Ra_floor_Za_def
  by (cases raw; simp, rule, simp add: φexpns, rule, simp add: φexpns)

lemma op_Ra_ceiling_Za_φapp[φoverload ceiling, φsynthesis 100]:
  ‹𝗉𝗋𝗈𝖼 op_Ra_ceiling_Za raw ⦃ x ⦂ 𝗏𝖺𝗅[raw] ℝ ⟼ ⌈x⌉ ⦂ 𝗏𝖺𝗅 ℤ ⦄›
  unfolding op_Ra_ceiling_Za_def
  by (cases raw; simp, rule, simp add: φexpns, rule, simp add: φexpns)


φoverloads to_real

definition op_Za_to_Ra :: ‹(VAL, VAL) proc'›
  where ‹op_Za_to_Ra rv =
    φM_getV aint V_aint.dest rv (λv.
    Return (φarg (V_areal.mk (real_of_int v))))›

declare [[
    overloaded_operator_in_synthesis ‹real_of_int›,
    overloaded_operator_in_synthesis ‹real_of_nat›
 ]]

lemma op_Za_to_Ra_φapp
  [φoverload to_real, φsynthesis for _ (100)
                                 and ‹x ⦂ 𝗏𝖺𝗅 ℤ› ⇒ ‹λv. real_of_int x ⦂ _› (1200)]:
  ‹ 𝗉𝗋𝗈𝖼 op_Za_to_Ra raw ⦃ x ⦂ 𝗏𝖺𝗅[raw] ℤ ⟼ real_of_int x ⦂ 𝗏𝖺𝗅 ℝ ⦄ ›
  unfolding op_Za_to_Ra_def
  by (cases raw; simp, rule, simp add: φexpns, rule, simp add: φexpns)

lemma op_Na_to_Ra_φapp
  [φoverload to_real, φsynthesis for _ (100)
                                 and ‹x ⦂ 𝗏𝖺𝗅 ℕ› ⇒ ‹λv. real_of_int x ⦂ _› (1200)]:
  ‹ 𝗉𝗋𝗈𝖼 op_Za_to_Ra raw ⦃ x ⦂ 𝗏𝖺𝗅[raw] ℕ ⟼ real_of_nat x ⦂ 𝗏𝖺𝗅 ℝ ⦄ ›
  ❴ op_Za_to_Ra ❵. .


end
