section ‹Specification of Monadic States›

definition StrictStateSpec :: "('ret::VALs φarg ⇒ rassn)
                          ⇒ (ABNM ⇒ rassn)
                          ⇒ 'ret comp set" ("!𝒮")
  where "!𝒮 T E = {s. case s of Success val x ⇒ x ∈ T val
                              | Abnormality val x ⇒ x ∈ E val
                              | Invalid ⇒ False
                              | NonTerm ⇒ False
                              | AssumptionBroken ⇒ False
                  }"

definition LooseStateSpec  :: "('ret::VALs φarg ⇒ rassn)
                          ⇒ (ABNM ⇒ rassn)
                          ⇒ 'ret comp set" ("𝒮")
  where  "𝒮 T E = {s. case s of Success val x ⇒ x ∈ T val
                              | Abnormality val x ⇒ x ∈ E val
                              | Invalid ⇒ False
                              | NonTerm ⇒ True
                              | AssumptionBroken ⇒ True
                  }"

lemma StrictStateSpec_expn[iff]:
        "Success vs x ∈ !𝒮 T E ≡ x ∈ T vs"
        "Abnormality v x ∈ !𝒮 T E ≡ x ∈ E v"
        "¬ (Invalid ∈ !𝒮 T E)"
        "¬ (NonTerm ∈ !𝒮 T E)"
        "¬ (AssumptionBroken ∈ !𝒮 T E)"
        "¬ {Invalid} ⊆ !𝒮 T E"
        "¬ {NonTerm} ⊆ !𝒮 T E"
        "¬ {AssumptionBroken} ⊆ !𝒮 T E"
  and LooseStateSpec_expn[iff]:
        "Success vs x ∈ 𝒮 T E ≡ x ∈ T vs"
        "Abnormality v x ∈ 𝒮 T E ≡ x ∈ E v"
        "¬ (Invalid ∈ 𝒮 T E)"
        "(NonTerm ∈ 𝒮 T E)"
        "(AssumptionBroken ∈ 𝒮 T E)"
        "¬ {Invalid} ⊆ 𝒮 T E"
        "{NonTerm} ⊆ 𝒮 T E"
        "{AssumptionBroken} ⊆ 𝒮 T E"
  by (simp_all add: StrictStateSpec_def LooseStateSpec_def)

lemma LooseStateSpec_expn' :
    "x ∈ 𝒮 T E ⟷ x = NonTerm
                 ∨ x = AssumptionBroken
                 ∨ (∃x' v. x = Success v x' ∧ x' ∈ T v)
                 ∨ (∃x' v. x = Abnormality v x' ∧ x' ∈ E v)"
  by (cases x) simp_all

lemma StrictStateSpec_elim[elim]:
    "s ∈ !𝒮 T E
⟹ (⋀x v. s = Success v x ⟹ x ∈ T v ⟹ C)
⟹ (⋀x v. s = Abnormality v x ⟹ x ∈ E v ⟹ C)
⟹ C" by (cases s) auto

lemma StrictStateSpec_intro[intro]:
    " x ∈ T v ⟹ Success v x ∈ !𝒮 T E"
    " x ∈ E a ⟹ Abnormality a x ∈ !𝒮 T E"
  by simp_all

lemma LooseStateSpec_E[elim]:
    "s ∈ 𝒮 T E
⟹ (⋀x v. s = Success v x ⟹ x ∈ T v ⟹ C)
⟹ (⋀x v. s = Abnormality v x ⟹ x ∈ E v ⟹ C)
⟹ (s = NonTerm ⟹ C)
⟹ (s = AssumptionBroken ⟹ C)
⟹ C"
  by (cases s) auto

lemma LooseStateSpec_I[intro]:
  "x ∈ T v ⟹ Success v x ∈ 𝒮 T E"
  "x ∈ E a ⟹ Abnormality a x ∈ 𝒮 T E"
  "NonTerm ∈ 𝒮 T E"
  "AssumptionBroken ∈ 𝒮 T E"
  by simp_all

lemma LooseStateSpec_upgrade:
  "s ∈ 𝒮 T E ⟹ s ≠ AssumptionBroken ⟹ s ≠ NonTerm ⟹ s ∈ !𝒮 T E"
  by (cases s) auto

lemma StrictStateSpec_degrade: "s ∈ !𝒮 T E ⟹ s ∈ 𝒮 T E" by (cases s) auto

lemma LooseStateSpec_introByStrict:
  "(s ≠ AssumptionBroken ⟹ s ≠ NonTerm ⟹ s ∈ !𝒮 T E) ⟹ s ∈ 𝒮 T E"
  by (cases s) auto

lemma StrictStateSpec_subset:
  ‹(⋀v. A v ⊆ A' v) ⟹ (⋀v. E v ⊆ E' v) ⟹ !𝒮 A E ⊆ !𝒮 A' E'›
  unfolding subset_iff StrictStateSpec_def by simp
lemma StrictStateSpec_subset'[intro]:
  ‹(⋀v. ∀s. s ∈ A v ⟶ s ∈ A' v) ⟹ (⋀v. ∀s. s ∈ E v ⟶ s ∈ E' v) ⟹ s ∈ !𝒮 A E ⟹ s ∈ !𝒮 A' E'›
  unfolding StrictStateSpec_def by (cases s; simp)

lemma LooseStateSpec_subset:
  ‹(⋀v. A v ⊆ A' v) ⟹ (⋀v. E v ⊆ E' v) ⟹ 𝒮 A E ⊆ 𝒮 A' E'›
  unfolding subset_iff LooseStateSpec_def by simp
lemma LooseStateSpec_subset'[intro]:
  ‹(⋀v. ∀s. s ∈ A v ⟶ s ∈ A' v) ⟹ (⋀v. ∀s. s ∈ E v ⟶ s ∈ E' v) ⟹ s ∈ 𝒮 A E ⟹ s ∈ 𝒮 A' E'›
  unfolding LooseStateSpec_def by (cases s; simp)


lemma LooseStateSpec_plus[iff]:
(*  ‹𝒮 (A + B) E   = 𝒮 A E + 𝒮 B E› *)
  ‹𝒮 X (λv. EA v + EB v) = 𝒮 X EA + 𝒮 X EB›
  unfolding set_eq_iff LooseStateSpec_def by simp_all
lemma StrictStateSpec_plus[iff]:
(*  ‹!𝒮 (A + B) E   = !𝒮 A E  + !𝒮 B E› *)
  ‹!𝒮 X (λv. EA v + EB v) = !𝒮 X EA + !𝒮 X EB›
  unfolding set_eq_iff StrictStateSpec_def by simp_all

lemma φCONSEQ:
   "𝗉𝗋𝗈𝖼 f ⦃ A  ⟼ B  ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E
⟹ A' 𝗌𝗁𝗂𝖿𝗍𝗌 A 𝖺𝗇𝖽 Any1
⟹ (⋀ret. B ret 𝗌𝗁𝗂𝖿𝗍𝗌 B' ret 𝖺𝗇𝖽 Any2)
⟹ (⋀ex.  E ex 𝗌𝗁𝗂𝖿𝗍𝗌 E' ex 𝖺𝗇𝖽 Any3)
⟹ 𝗉𝗋𝗈𝖼 f ⦃ A' ⟼ B' ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E' "
  unfolding φProcedure_def View_Shift_def subset_iff
  apply clarsimp
  by (smt (verit, del_insts) LooseStateSpec_expn')

subsection ‹Helper Rules›

lemma φframe0:
  "𝗉𝗋𝗈𝖼 f ⦃ A ⟼ B ⦄ ⟹ 𝗉𝗋𝗈𝖼 f ⦃ R * A ⟼ λret. R * B ret ⦄"
  using φframe[where E=0, simplified, folded zero_fun_def] .

lemma φCONSEQ'E:
   "(⋀v. E v 𝗌𝗁𝗂𝖿𝗍𝗌 E' v 𝖺𝗇𝖽 P3)
⟹ 𝗉𝗋𝗈𝖼 f ⦃ A  ⟼ B  ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E
⟹ 𝗉𝗋𝗈𝖼 f ⦃ A ⟼ B ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E' "
  using φCONSEQ view_shift_refl by blast

lemmas φCONSEQ'E0 = φCONSEQ'E[OF view_shift_0, unfolded zero_fun_eta]

subsubsection ‹Case Analysis›

lemma φCASE:
  ‹ 𝗉𝗋𝗈𝖼 f ⦃ A ⟼ C ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E
⟹ 𝗉𝗋𝗈𝖼 f ⦃ B ⟼ C ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E
⟹ 𝗉𝗋𝗈𝖼 f ⦃ A + B ⟼ C ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E›
  unfolding φProcedure_def
  by (simp add: distrib_left)

lemma φCASE_VS:
  ‹ A 𝗌𝗁𝗂𝖿𝗍𝗌 Y 𝖺𝗇𝖽 P1
⟹ B 𝗌𝗁𝗂𝖿𝗍𝗌 Y 𝖺𝗇𝖽 P2
⟹ B + A 𝗌𝗁𝗂𝖿𝗍𝗌 Y 𝖺𝗇𝖽 P2 ∨ P1›
  unfolding View_Shift_def
  by (simp add: distrib_left)

lemma φCASE_IMP:
  ‹ A 𝗂𝗆𝗉𝗅𝗂𝖾𝗌 Y 𝖺𝗇𝖽 P1
⟹ B 𝗂𝗆𝗉𝗅𝗂𝖾𝗌 Y 𝖺𝗇𝖽 P2
⟹ B + A 𝗂𝗆𝗉𝗅𝗂𝖾𝗌 Y 𝖺𝗇𝖽 P2 ∨ P1›
  unfolding Imply_def
  by (simp add: distrib_left)


subsubsection ‹Normalization in Precondition›

lemma norm_precond_conj:
  "(𝗉𝗋𝗈𝖼 f ⦃ T 𝗌𝗎𝖻𝗃 P ⟼ Y ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E) = (P ⟶ 𝗉𝗋𝗈𝖼 f ⦃ T ⟼ Y ⦄ 𝗍𝗁𝗋𝗈𝗐𝗌 E )"
  unfolding φProcedure_def by (simp add: φexpns) blast

lemmas norm_precond_conj_metaeq[unfolded atomize_eq[symmetric]] = norm_precond_conj

  unfolding φProcedure_def by (simp add: φexpns) blast
