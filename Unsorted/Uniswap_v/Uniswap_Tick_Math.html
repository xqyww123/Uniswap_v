Theory Uniswap_Tick_Math

theory Uniswap_Tick_Math
  imports Uniswap_Common HOL.Transcendental
begin

definition ‹FACTOR = (1.0001::real)›

lemma FACTOR_LG_1: ‹1 < FACTOR› unfolding FACTOR_def by simp

definition price_of :: ‹int ⇒ real›
  where price_of_def': ‹price_of tick = sqrt (FACTOR powr (of_int tick))›

lemma price_of_def: ‹price_of t = FACTOR powr (of_int t / 2)›
  unfolding price_of_def'
  by (simp add: powr_half_sqrt[symmetric] powr_powr)

hide_fact price_of_def'

lemma price_of_L0[simp]:
  ‹0 < price_of x›
  using FACTOR_LG_1 price_of_def by auto

lemma price_of_mono': ‹mono price_of›
  unfolding price_of_def mono_on_def
  by (simp add: FACTOR_LG_1)

lemma price_of_smono': ‹strict_mono price_of›
  unfolding price_of_def strict_mono_on_def
  by (simp add: FACTOR_LG_1)

lemma price_of_mono:
  ‹price_of x ≤ price_of y ⟷ x ≤ y›
  using price_of_mono'
  by (simp add: price_of_smono' strict_mono_less_eq) 

lemma price_of_smono:
  ‹price_of x < price_of y ⟷ x < y›
  using price_of_smono' using strict_mono_less by blast

definition ‹tick_of_price p = (@t. price_of t ≤ p ∧ p < price_of (t+1)) ›


lemma tick_of_price[simp]: ‹tick_of_price (price_of t) = t›
  unfolding tick_of_price_def
  by (smt (z3) price_of_smono some_eq_imp)

lemma price_of_tick:
  assumes ‹0 < p›
  shows ‹price_of (tick_of_price p) ≤ p ∧ p < price_of (tick_of_price p + 1)›
proof -
  have ‹∃t. p < FACTOR powr t›
    by (metis FACTOR_LG_1 dual_order.strict_trans floor_log_eq_powr_iff linorder_neqE_linordered_idom zero_less_one)
  then have ‹∃t. p < price_of t›
    unfolding price_of_def
    by (meson FACTOR_LG_1 dual_order.strict_trans ex_less_of_int less_divide_eq_numeral1(1) powr_less_cancel_iff)
  moreover have ‹∃t. FACTOR powr t ≤ p›
    using ‹0 < p› FACTOR_LG_1 floor_log_eq_powr_iff by blast
  then have ‹∃t. price_of t ≤ p›
    proof -
      have ‹⋀x. ∃t. real_of_int t / 2 < x› by (simp add: ex_of_int_less)
      then show ?thesis
      unfolding price_of_def
      by (meson FACTOR_LG_1 ‹∃t. FACTOR powr t ≤ p› dual_order.trans less_le_not_le powr_le_cancel_iff)
    qed
  ultimately have X: ‹(∃t. price_of t ≤ p ∧ p < price_of (t+1))›
    proof -
    obtain t1 where t1: ‹price_of t1 ≤ p› using ‹∃t. price_of t ≤ p› by blast
    obtain t2 where t2: ‹p < price_of t2› using ‹∃t. p < price_of t› by blast
    have le: ‹t1 < t2› using t1 t2 price_of_smono by fastforce
    thm int_gr_induct[OF le]
    have ‹p < price_of t2 ⟶ ?thesis›
    proof (induct rule: int_gr_induct[OF le])
      case 1
      then show ?case
        using t1 by blast
    next
      case (2 i)
      then show ?case
        by force
    qed
    then show ?thesis
      using t2 by blast
  qed
  then show ?thesis
    unfolding tick_of_price_def
    by (metis (no_types, lifting) someI_ex)
qed

lemma tick_of_price_LE_mono:
  ‹0 < a ⟹ a ≤ b ⟹ tick_of_price a ≤ tick_of_price b›
  by (smt (verit) price_of_smono price_of_tick)
  


text ‹As we are not planning verifying the numeric calculation and its precision (in a short term),
  we declare them directly as assumed axioms.›

debt_axiomatization
    getSqrtRatioAtTick :: ‹(VAL,VAL) proc'› and
    getTickAtSqrtRatio :: ‹(VAL,VAL) proc'›
  where getSqrtRatioAtTick_φapp:
            ‹𝗉𝗋𝗈𝖼 getSqrtRatioAtTick v ⦃ t ⦂ 𝗏𝖺𝗅[v] Tick ⟼ price_of t ⦂ 𝗏𝖺𝗅 ℝ ⦄›
    and getTickAtSqrtRatio_φapp:
            ‹𝗉𝗋𝖾𝗆𝗂𝗌𝖾 0 < p
         ⟹ 𝗉𝗋𝗈𝖼 getTickAtSqrtRatio v ⦃ p ⦂ 𝗏𝖺𝗅[v] ℝ ⟼ tick_of_price p ⦂ 𝗏𝖺𝗅 Tick ⦄›

abbreviation ‹MIN_PRICE ≡ price_of MIN_TICK›
abbreviation ‹MAX_PRICE ≡ price_of MAX_TICK›

lemma [φreason 1000]:
  ‹ Is_Literal X
⟹ Is_Literal (price_of X)›
  unfolding Is_Literal_def .

proc (nodef) MIN_PRICE[φsynthesis 1100]:
  input ‹Void›
  output ‹MIN_PRICE ⦂ 𝗏𝖺𝗅 ℝ›
  ❴ op_const_areal[where x=‹MIN_PRICE›] ❵. .

proc (nodef) MAX_PRICE[φsynthesis 1100]:
  input ‹Void›
  output ‹MAX_PRICE ⦂ 𝗏𝖺𝗅 ℝ›
  ❴ op_const_areal[where x=‹MAX_PRICE›] ❵. .

thm price_of_tick

lemma
  ‹0 < p ⟹ p < price_of A ⟹ tick_of_price p < A›
  using price_of_smono price_of_tick by fastforce

lemma less_MAX_PRICE_less_MAX_TICK:
  ‹0 < p ⟹ p < MAX_PRICE ⟹ tick_of_price p < MAX_TICK›
  using price_of_smono price_of_tick by fastforce

lemma tick_of_price_LT_MAX_price_LT_MAX:
  ‹tick_of_price p < MAX_TICK ⟹ p < MAX_PRICE›
  by (smt (verit, ccfv_SIG) price_of_L0 price_of_smono price_of_tick)
