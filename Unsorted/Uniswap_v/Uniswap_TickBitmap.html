Theory Uniswap_TickBitmap

theory Uniswap_TickBitmap
  imports Uniswap_Common
begin

(*will be a good case to test our φ-type-fiction conversion.
Get fiction relation from φ-type relation.*)

type_synonym tick_map = ‹tick ⇒ bool›

locale Tickmap_spec =
  fixes Tickmap :: ‹(fiction, tick_map) φ›
    and op_get_tickmap :: ‹(VAL, VAL) proc'›
    and nextInitializedTickWithinOneWord :: ‹(VAL × VAL, VAL × VAL) proc'›
  assumes op_get_tickmap_φapp:
    ‹ 𝗉𝗋𝗈𝖼 op_get_tickmap ri ⦃ m ⦂ Tickmap❟ i ⦂ 𝗏𝖺𝗅[ri] Tick
                          ⟼ m ⦂ Tickmap❟ m i ⦂ 𝗏𝖺𝗅 𝔹⦄ ›
  and nextInitializedTickWithinOneWord_φapp:
    (*we don't consider TickSpacing in this abstract specification.*)
    ‹ 𝗉𝗋𝗈𝖼 nextInitializedTickWithinOneWord (rlte❙, ri)
          ⦃ m ⦂ Tickmap❟ i ⦂ 𝗏𝖺𝗅[ri] ℤ❟ lte ⦂ 𝗏𝖺𝗅[rlte] 𝔹
        ⟼ m ⦂ Tickmap❟ j ⦂ 𝗏𝖺𝗅 ℤ❟ m j ⦂ 𝗏𝖺𝗅 𝔹 𝗌𝗎𝖻𝗃 j.
             (if lte then j ≤ i ∧ (∀k ∈ {j<..i}. ¬ m k)
                     else i < j ∧ (∀k ∈ {i<..<j}. ¬ m k)) ⦄ ›

end
