lemma fee0_plus_homo[simp]:
  ‹growth.fee0 (a + b) = growth.fee0 a + growth.fee0 b›
  by (cases a; cases b; simp)

lemma fee0_sub_homo[simp]:
  ‹growth.fee0 (a - b) = growth.fee0 a - growth.fee0 b›
  by (cases a; cases b; simp)

lemma fee0_sum:
  ‹growth.fee0 (sum f S) = sum (growth.fee0 o f) S›
  by (metis add.right_neutral add_diff_cancel_left' fee0_plus_homo sum_comp_morphism)

lemma fee1_plus_homo[simp]:
  ‹growth.fee1 (a + b) = growth.fee1 a + growth.fee1 b›

lemma fee1_sub_homo[simp]:
  ‹growth.fee1 (a - b) = growth.fee1 a - growth.fee1 b›
  by (cases a; cases b; simp)

lemma fee1_sum:
  ‹growth.fee1 (sum f S) = sum (growth.fee1 o f) S›
  by (metis add.right_neutral add_diff_cancel_left' fee1_plus_homo sum_comp_morphism)

  where [φdefs]: ‹Address a = (a ⦂ ℤ)›

lemma [φreason 1200]:
  ‹ 𝗍𝗁𝗋𝖾𝗌𝗁𝗈𝗅𝖽 1
⟹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 i = j
⟹ i ⦂ ℤ 𝗂𝗆𝗉𝗅𝗂𝖾𝗌 j ⦂ Address›
  ❴ constructφ ‹i ⦂ Address› ❵. .

lemma [φreason 1200]:
  ‹ i ⦂ ℤ 𝗂𝗆𝗉𝗅𝗂𝖾𝗌 i ⦂ Address @action to Address› ❴ ❵. .

lemma [φreason 1200]:
  ‹ 𝗍𝗁𝗋𝖾𝗌𝗁𝗈𝗅𝖽 1
⟹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 i = j
⟹ i ⦂ Address 𝗂𝗆𝗉𝗅𝗂𝖾𝗌 j ⦂ ℤ ›
  ❴ destructφ _ ❵. .
