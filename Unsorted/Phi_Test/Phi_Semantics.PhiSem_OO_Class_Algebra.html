Theory Phi_Semantics.PhiSem_OO_Class_Algebra

theory PhiSem_OO_Class_Algebra
  imports PhiSem_Base
begin

section ‹Algebraic Model of Class dependency›

datatype 'members class_algebra =
  Class (name: string) (parents: ‹'members class_algebra list›) (members: 'members)

text ‹If two classes having the same name but different parents or members,
  they are merely two different classes having the same name.

  If two classes having the identical parents and members but different names,
  they are still two different classes despite of similarity.

  The name may be used as a syntactical identifier in language implementations, but the equity of
    the names in the semantics never means the equity of the classes.

  The member is generic, which may contain types of fields, table of virtual functions.
›

hide_const (open) name parents members

fun self_and_parents_of :: "'a class_algebra ⇒ 'a class_algebra set" 
  where ‹self_and_parents_of (Class name parents fields) =
            insert (Class name parents fields) (⋃ (self_and_parents_of ` set parents))›

instantiation class_algebra :: (type) order begin

definition ‹less_eq_class_algebra C1 C2 ⟷ C2 ∈ self_and_parents_of C1›
definition ‹less_class_algebra (C1::'a class_algebra) C2 ⟷ C1 ≤ C2 ∧ C1 ≠ C2›

instance proof
  fix x y z :: ‹'a class_algebra›

  have ‹⋀x y S'. x ∈ self_and_parents_of y ⟹ x ≠ y
            ⟹ size_class_algebra S' x < size_class_algebra S' y›
  subgoal for x y S'
  apply (induct y arbitrary: x)
  subgoal for name parents fields x
    apply clarsimp
    subgoal for z
      apply (cases ‹x = z›)
       apply (metis not_add_less1 not_less_eq size_list_estimation)
      by (simp add: less_Suc_eq size_list_estimation trans_less_add1) . . .
  then have t1: ‹⋀x y. y ∈ self_and_parents_of x ∧ x ∈ self_and_parents_of y ⟹ x = y›
    by (metis order_less_asym)
  then  show ‹(x < y) = (x ≤ y ∧ ¬ y ≤ x)›
    apply (simp_all add: less_eq_class_algebra_def less_class_algebra_def)
    by blast

  show ‹ x ≤ y ⟹ y ≤ x ⟹ x = y›
    by (simp add: less_eq_class_algebra_def less_class_algebra_def t1)

  show ‹x ≤ x›
    by (cases x; simp add: less_eq_class_algebra_def less_class_algebra_def)

  show ‹x ≤ y ⟹ y ≤ z ⟹ x ≤ z›
    unfolding less_eq_class_algebra_def less_class_algebra_def
    by (induct x, auto)
qed
end

lemma [simp]: ‹x ∈ self_and_parents_of x›
  by (cases x, simp)

lemma [simp]: ‹Class name (x # L) members ≠ x›
proof
  assume A: ‹Class name (x # L) members = x›
  then have t: ‹⋀S. size_class_algebra S (Class name (x # L) members) = size_class_algebra S x› by simp
  then show False
    by (metis A add_Suc_right class_algebra.size(2) list.set_intros(1) not_add_less1 not_less_eq size_list_estimation)
qed

lemma prop_subclass_1[intro]: ‹Class name (x#L) members < x›
  unfolding less_class_algebra_def less_eq_class_algebra_def by simp

lemma subclass_1[intro]: ‹Class name (x#L) members ≤ x›
  unfolding less_class_algebra_def less_eq_class_algebra_def by simp

lemma subclass_0[intro]: ‹x ≤ x› for x :: ‹'a class_algebra› by simp
