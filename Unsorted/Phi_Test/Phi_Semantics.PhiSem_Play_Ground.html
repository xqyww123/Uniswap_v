fun fib :: ‹nat ⇒ nat› where
  ‹fib 0 = 1› | ‹fib (Suc 0) = 1› | ‹fib n = fib (n-1) + fib (n-2)›

proc FIB:
  input ‹𝗏𝖺𝗅 n ⦂ ℕ›
  output ‹𝗏𝖺𝗅 fib n ⦂ ℕ›
  is [recursive n]
  is [routine]
❴
  if ‹$n ≤ 1› ❴ 1 ❵ ❴
    FIB (‹$n - 1›) add (FIB (‹$n - 2›))
  ❵
❵.

proc FIB2:
  input ‹𝗏𝖺𝗅 n ⦂ ℕ(8)›
  output ‹𝗏𝖺𝗅 fib n ⦂ ℕ⇧r(32)›
  is [recursive n]
❴
  if ‹$n ≤ 1›
  ❴ ‹1 ⦂ ℕ⇧r(32)› ❵
  ❴ FIB2 (‹$n - 1›) add (FIB2 (‹$n - 2›)) ❵
❵.

thm FIB2_def
