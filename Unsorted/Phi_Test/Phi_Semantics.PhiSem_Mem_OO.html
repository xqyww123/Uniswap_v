paragraph ‹Dispose›

lemma op_obj_dispose:
  ‹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 ref ≠ Nil
⟹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 dom fields = dom (class.fields_of cls)
⟹ 𝗉𝗋𝗈𝖼 op_obj_dispose cls rawv
    ⦃ to_share o fields ⦂ obj: ref ❙→ Identity ❟ ref ⦂ Val rawv (Ref cls) ⟼ λret. Void ⦄›
  unfolding op_obj_dispose_def Premise_def
  apply (rule φM_getV_ref)
  apply (rule φSEQ[where B=‹λ_. to_share ∘ fields ⦂ obj: ref ❙→ Identity›])
  apply (clarsimp simp add: φexpns zero_set_def φProcedure_Hybrid_DL del: subsetI)
  apply (rule RES.Objs.φR_get_res, simp, simp add: dom1_def)
  subgoal premises prems for r res proof -
    have t1: ‹object_ref.class ref = cls›
      by (metis object_ref.collapse of_class.simps(1) prems(1) prems(3))
    have t3: ‹fields = Map.empty ⟹ class.fields_of cls = Map.empty› 
      subgoal premises prem proof -
        have ‹dom fields = {}› by (simp add: prem)
        then have ‹dom (class.fields_of cls) = {}› using prems(2) by simp
        then show ?thesis by fastforce
      qed .
    have t4: ‹r ## FIC.OO_share.mk (1(ref := to_share ∘ fields))›
      using prems(6) by blast
    have t2: ‹RES.Objs.get res ref = 1 ⟹ class.fields_of cls = 1›
      unfolding one_fun_def one_option_def
      apply (cases ‹fields = Map.empty›)
      using t3 apply blast
      using FIC.OO_share.partial_implies[where x=‹1(ref := fields)› and n=1, simplified,
            OF ‹r ∈ FIC.SPACE›, OF t4, OF ‹𝗌𝗍𝖺𝗍𝖾 res 𝗂𝗌 _›]
            nonsepable_partial_map_subsumption_L2
      by (metis domIff map_le_def)
    show ?thesis by (simp add: t1 t2 prems Return_def det_lift_def)
  qed
  apply (rule FIC.OO_share.φR_dispose_res[where P=‹λ_. True›])
  apply (clarsimp simp add: one_fun_def)
  apply (cases ref; simp)
  using RES.Objs.get_res_Valid[simplified Valid_Objs_def, simplified]
    RES.Objs.raw_unit_assertion_implies'[where f=fields]
  by (smt (z3) domIff map_le_antisym map_le_def)
