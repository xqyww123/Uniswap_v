Theory Virtual_Datatype.Virtual_Datatype

theory Virtual_Datatype
  imports "Phi_Statespace.StateFun"
  keywords "virtual_datatype" :: thy_decl
begin

locale "virtual_datatype" =
  fixes CONS_OF :: "'value ⇒ 'CONS_NAME"
begin

lemma [simp]: "CONS_OF x ≠ CONS_OF y ≡ True ⟹ x ≠ y ≡ True"
  unfolding atomize_eq by blast

end

datatype ('CONS_NAME,'REP,'ABS) Field =
  Field (name: 'CONS_NAME) (project: "'REP ⇒ 'ABS") (inject: "'ABS ⇒ 'REP")

hide_const (open) name project inject

definition lift_Field :: ‹('ABS ⇒ 'ABS')
                       ⇒ ('ABS' ⇒ 'ABS)
                       ⇒ ('CONS_NAME,'REP,'ABS) Field
                       ⇒ ('CONS_NAME,'REP,'ABS') Field›
  where ‹lift_Field f g field = (case field of Field name projector injector ⇒
            Field name (f o projector) (injector o g))›

lemma lift_Field[simp]:
  ‹lift_Field f g (Field name projector injector) = Field name (f o projector) (injector o g)›
  unfolding lift_Field_def by simp


locale VDT_field =
  fixes field :: "('CONS_NAME,'value,'a) Field"
  assumes dest_mk[simp]: "Field.project field (Field.inject field x) = x"
begin

abbreviation "mk ≡ Field.inject field"
abbreviation "dest ≡ Field.project field"
abbreviation "name ≡ Field.name field"

lemma dest_mk_id[simp]: "dest o mk = id"
  using dest_mk destr_contstr_comp_id by blast

lemma dest_mk_cancle[simp]: "f o dest o mk = f"
  by fastforce

lemma mk_inj[iff]: "mk x = mk y ⟷ x = y"
  by (metis dest_mk)

definition "domain = {v. (∃x. v = mk x)}"

lemma in_domainE[elim!]: "v ∈ domain ⟹ (⋀x. v = mk x ⟹ C) ⟹ C"
  using domain_def by blast

lemma mk_in_domain[simp,intro]: "mk x ∈ domain" using domain_def by blast

end

  ‹(⋀x. f (g x) = x) ⟹ VDT_field field ⟹ VDT_field (lift_Field f g field)›
  unfolding VDT_field_def
  by (cases field; simp)*)


locale field_entry =
  "virtual_datatype" CONS_OF + VDT_field field
  for CONS_OF :: "'value ⇒ 'CONS_NAME"
  and field :: "('CONS_NAME,'value,'a) Field"
+ assumes name_of[simp]: "CONS_OF (Field.inject field x) = Field.name field"
begin

lemma not_constructor_not_instance[simp]:
  ‹CONS_OF v ≠ name ⟹ v ∉ domain ≡ True›
  unfolding atomize_eq using name_of by blast

lemma in_domain_cons[simp]: "v ∈ domain ⟹ CONS_OF v = name"
  using name_of domain_def by force

lemma in_domain_mk_dest[simp]: "v ∈ domain ⟹ mk (dest v) = v"
  using domain_def dest_mk by fastforce

end
